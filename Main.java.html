import java.util.Scanner;
public class Main
{
    public static int result(String word){
        if(word.charAt(0) == '+') return 1;
        else if(word.charAt(0) == '-') return 2;
        else if(word.charAt(0) == '*') return 3;
        else if(word.charAt(0) == '/') return 4;
        else return 5;
    }
    public static String calc(String input){
        int first, second;
        String[] words = input.split(" ");
        if(words.length != 3){
            System.out.println("Исключение!");
            return "";
        }
        for(int i = 0; i < words[0].length(); ++i){
            if(!Character.isDigit(words[0].charAt(i))){
                System.out.println("Исключение!");
                return "";
            }
        }
        first = Integer.parseInt(words[0]);
        for(int i = 0; i < words[2].length(); ++i){
            if(!Character.isDigit(words[2].charAt(i))){
                System.out.println("Исключение!");
                return "";
            }
        }
        second = Integer.parseInt(words[2]);
        if(result(words[1]) == 1) return String.valueOf(first + second);
        else if(result(words[1]) == 2) return String.valueOf(first - second);
        else if(result(words[1]) == 3) return String.valueOf(first * second);
        else if(result(words[1]) == 4) return String.valueOf(first / second);
        else{
            System.out.println("Исключение!");
            return "";
        }
    }
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        System.out.print("Введите пример: ");
        String text = in.nextLine();
        System.out.println(calc(text));
    }
}
